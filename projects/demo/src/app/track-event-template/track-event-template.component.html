Advanced events

Change value and blur this input to trigger event 
  <mat-form-field>
    <mat-label>Change value and blur this input to trigger event</mat-label>
    <input type="text"
           matInput
           matomoTracker
           #tracker="matomo"
           [matomoCategory]="category"
           [matomoAction]="action"
           [matomoName]="name"
           [matomoValue]="value"
           (change)="tracker.trackEvent()">
